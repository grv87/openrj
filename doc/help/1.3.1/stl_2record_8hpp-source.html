00388 inline ORJField const *record::find_field_(string_t const &name) const
00389 {
00390     // We do a boring linear search.
00391     ORJField const  *const  b           =   &m_record->fields[0];
00392     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00393     ORJField const          *it;
00394     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00395 
00396     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00397     {
00398         // Ordered, so we can do a binary search
00399         it  =   std::lower_bound(b, e, name, same_name());
00400 
00401         if(e != it)
00402         {
00403 #if defined(__STLSOFT_COMPILER_IS_GCC) && \
00404     __GNUC__ < 3
00405             // This copes with the non-standard GCC 2.x standard library, but
00406             // will also work with a good one (albeit less efficiently).
00407             if(0 == string_t((*it).name.ptr, (*it).name.len).compare(name))
00408 #else /* ? compiler */
00409             if(0 != name.compare(0, (*it).name.len, (*it).name.ptr))
00410 #endif /* compiler */
00411             {
00412                 it = e;
00413             }
00414         }
00415     }
00416     else
00417     {
00418         // Unordered, so we must do a linear search
00419         it  =   std::find_if(b, e, same_name(name));
00420     }
00421 
00422     if(e == it)
00423     {
00424         it = NULL;
00425     }
00426 
00427     return it;
00428 }

00430 inline size_t record::count_fields_(string_t const &name) const
00431 {
00432     ORJField const  *const  b           =   &m_record->fields[0];
00433     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00434     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00435 
00436     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00437     {
00438         // Ordered, so we can do a binary search
00439         ORJField const  *lower  =   std::lower_bound(b, e, name, same_name());
00440         ORJField const  *upper  =   std::upper_bound(b, e, name, same_name());
00441 
00442         return static_cast<size_t>(std::distance(lower, upper)); // distance() return ptrdiff_t
00443     }
00444     else
00445     {
00446         // Unordered, so we must do a linear search
00447         return static_cast<size_t>(std::count_if(b, e, same_name(name))); // some impls do not return size_t
00448     }
00449 }
