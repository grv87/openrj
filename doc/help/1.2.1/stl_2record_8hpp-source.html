00385 inline ORJField const *record::find_field_(string_t const &name) const
00386 {
00387     // We do a boring linear search.
00388     ORJField const  *const  b           =   &m_record->fields[0];
00389     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00390     ORJField const          *it;
00391     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00392 
00393     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00394     {
00395         // Ordered, so we can do a binary search
00396         it  =   std::lower_bound(b, e, name, same_name());
00397 
00398         if(e != it)
00399         {
00400 #if defined(__STLSOFT_COMPILER_IS_GCC) && \
00401     __GNUC__ < 3
00402             // This copes with the non-standard GCC 2.x standard library, but
00403             // will also work with a good one (albeit less efficiently).
00404             if(0 == string_t((*it).name.ptr, (*it).name.len).compare(name))
00405 #else /* ? compiler */
00406             if(0 != name.compare(0, (*it).name.len, (*it).name.ptr))
00407 #endif /* compiler */
00408             {
00409                 it = e;
00410             }
00411         }
00412     }
00413     else
00414     {
00415         // Unordered, so we must do a linear search
00416         it  =   std::find_if(b, e, same_name(name));
00417     }
00418 
00419     if(e == it)
00420     {
00421         it = NULL;
00422     }
00423 
00424     return it;
00425 }

00427 inline size_t record::count_fields_(string_t const &name) const
00428 {
00429     ORJField const  *const  b           =   &m_record->fields[0];
00430     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00431     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00432 
00433     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00434     {
00435         // Ordered, so we can do a binary search
00436         ORJField const  *lower  =   std::lower_bound(b, e, name, same_name());
00437         ORJField const  *upper  =   std::upper_bound(b, e, name, same_name());
00438 
00439         return static_cast<size_t>(std::distance(lower, upper)); // distance() return ptrdiff_t
00440     }
00441     else
00442     {
00443         // Unordered, so we must do a linear search
00444         return static_cast<size_t>(std::count_if(b, e, same_name(name))); // some impls do not return size_t
00445     }
00446 }
