00448 inline ORJField const *record::find_field_(string_t const &name) const
00449 {
00450     // We do a boring linear search.
00451     ORJField const  *const  b           =   &m_record->fields[0];
00452     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00453     ORJField const          *it;
00454     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00455 
00456     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00457     {
00458         // Ordered, so we can do a binary search
00459         it  =   std::lower_bound(b, e, name, same_name());
00460 
00461         if(e != it)
00462         {
00463 #if defined(__STLSOFT_COMPILER_IS_GCC) && \
00464     __GNUC__ < 3
00465             // This copes with the non-standard GCC 2.x standard library, but
00466             // will also work with a good one (albeit less efficiently).
00467             if(0 == string_t((*it).name.ptr, (*it).name.len).compare(name))
00468 #else /* ? compiler */
00469             if(0 != name.compare(0, (*it).name.len, (*it).name.ptr))
00470 #endif /* compiler */
00471             {
00472                 it = e;
00473             }
00474         }
00475     }
00476     else
00477     {
00478         // Unordered, so we must do a linear search
00479         it  =   std::find_if(b, e, same_name(name));
00480     }
00481 
00482     if(e == it)
00483     {
00484         it = NULL;
00485     }
00486 
00487     return it;
00488 }

00490 inline size_t record::count_fields_(string_t const &name) const
00491 {
00492     ORJField const  *const  b           =   &m_record->fields[0];
00493     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00494     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00495 
00496     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00497     {
00498         // Ordered, so we can do a binary search
00499         ORJField const  *lower  =   std::lower_bound(b, e, name, same_name());
00500         ORJField const  *upper  =   std::upper_bound(b, e, name, same_name());
00501 
00502         return static_cast<size_t>(std::distance(lower, upper)); // distance() return ptrdiff_t
00503     }
00504     else
00505     {
00506         // Unordered, so we must do a linear search
00507         return static_cast<size_t>(std::count_if(b, e, same_name(name))); // some impls do not return size_t
00508     }
00509 }
