00480 inline ORJField const* record::find_field_(string_t const &name) const
00481 {
00482     // We do a boring linear search.
00483     ORJField const  *const  b           =   &m_record->fields[0];
00484     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00485     ORJField const          *it;
00486     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00487 
00488     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00489     {
00490         // Ordered, so we can do a binary search
00491         it  =   std::lower_bound(b, e, name, same_name());
00492 
00493         if(e != it)
00494         {
00495 #if defined(STLSOFT_COMPILER_IS_GCC) && \
00496     __GNUC__ < 3
00497             // This copes with the non-standard GCC 2.x standard library, but
00498             // will also work with a good one (albeit less efficiently).
00499             if(0 == string_t((*it).name.ptr, (*it).name.len).compare(name))
00500 #else /* ? compiler */
00501             if(0 != name.compare(0, (*it).name.len, (*it).name.ptr))
00502 #endif /* compiler */
00503             {
00504                 it = e;
00505             }
00506         }
00507     }
00508     else
00509     {
00510         // Unordered, so we must do a linear search
00511         it  =   std::find_if(b, e, same_name(name));
00512     }
00513 
00514     if(e == it)
00515     {
00516         it = NULL;
00517     }
00518 
00519     return it;
00520 }

00522 inline size_t record::count_fields_(string_t const &name) const
00523 {
00524     ORJField const  *const  b           =   &m_record->fields[0];
00525     ORJField const  *const  e           =   &m_record->fields[m_record->numFields];
00526     ORJDatabase const       *database   =   ORJ_Record_GetDatabaseA(m_record);
00527 
00528     if(ORJ_FLAG_ORDERFIELDS == (database->flags & ORJ_FLAG_ORDERFIELDS))
00529     {
00530         // Ordered, so we can do a binary search
00531         ORJField const  *lower  =   std::lower_bound(b, e, name, same_name());
00532         ORJField const  *upper  =   std::upper_bound(b, e, name, same_name());
00533 
00534         return static_cast<size_t>(std::distance(lower, upper)); // distance() return ptrdiff_t
00535     }
00536     else
00537     {
00538         // Unordered, so we must do a linear search
00539         return static_cast<size_t>(std::count_if(b, e, same_name(name))); // some impls do not return size_t
00540     }
00541 }
